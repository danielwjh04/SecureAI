# SecureSG — optional ML wheels (opt-in, real-model validation only).
#
# These power the real semantic layer: the Qwen3 judge (GuardFormer) and the
# MiniLM intent-drift embeddings. They are heavy native wheels (a C++ build for
# llama-cpp, a large torch download) and are imported lazily behind swappable
# provider seams, so neither the package nor the non-gated test suite needs them
# — SecureSG runs deterministic-only without them. Install on top of the base
# requirements only when validating against real models:
#
#     pip install -r requirements.txt -r requirements-ml.txt
#
# then fetch the weights (python -m scripts.fetch_model) and set SECURESG_MODEL_PATH.

# GuardFormer judge — SP3 (semantic screening layer).

huggingface-hub>=0.24      # one-time weights download (scripts/fetch_model.py)

# Warden embeddings — SP4 (intent-to-action drift detection).
sentence-transformers>=3.0  # all-MiniLM-L6-v2 sentence embeddings (CPU)
